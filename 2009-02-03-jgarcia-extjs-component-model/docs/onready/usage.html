Usage examples

Inline:

Ext.onReady(function() {
	
    Ext.MessageBox.alert('Hello!', 'Welcome to Ext JS!');

});

Scoped:

var mySingleton = function() {
	
    var title = 'Hello!';
    this.msg = "Welcome to Ext JS!",
    
    return {
    	init : function() {
            Ext.MessageBox.alert(title, this.msg);
        }
    }
}();

Ext.onReady(mySingleton.init, mySingleton);
